Docker Compose is a tool for defining and running multi-container Docker applications using a docker-compose.yml file.

ğŸ“ Sample docker-compose.yml File

version: '3'
services:
  web:
    image: nginx
    ports:
      - "8080:80"

  app:
    build: .
    volumes:
      - .:/app
    depends_on:
      - db

  db:
    image: mysql:5.7
    environment:
      MYSQL_ROOT_PASSWORD: root
