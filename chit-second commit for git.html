4️⃣ Live Practical Examples

🚀 Run Nginx

docker run -d -p 8080:80 nginx

🛠️ Dockerfile for Node App

FROM node:18
WORKDIR /app
COPY . .
RUN npm install
CMD ["node", "app.js"]
docker build -t my-node-app .
docker run -d -p 3000:3000 my-node-app

⚙️ Docker Compose (Node + Mongo)

version: "3"
services:
  backend:
    build: .
    ports:
      - "3000:3000"
    depends_on:
      - mongo

  mongo:
    image: mongo
    ports:
      - "27017:27017"
docker-compose up --build

📂 Volume Mount

docker run -v mydata:/data busybox sh -c "echo Hello > /data/hello.txt"
docker run -v mydata:/data busybox cat /data/hello.txt

🌐 Custom Network

docker network create mynet
docker run -d --name db --network mynet mongo
docker run -it --network mynet mongo mongo --host db
